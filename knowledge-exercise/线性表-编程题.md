=== 知识点ID: 11
=== 知识点名称: 顺序表

## PROGRAMMING
题型: PROGRAMMING
难度: EASY
类别: BASIC
题目: 顺序表插入
描述:
给定一个长度为 n 的顺序表和一个待插入元素，请在指定位置插入该元素，并输出插入后的顺序表。

输入说明：
第一行输入整数 n，表示顺序表当前元素个数。
第二行输入 n 个整数，表示顺序表中的元素。
第三行输入两个整数 pos 和 x，表示将元素 x 插入到第 pos 个位置。

输出说明：
输出插入后的顺序表元素，元素之间用一个空格分隔。
若 pos 不在 1 到 n+1 范围内，输出 ERROR。

样例输入：
5
1 2 4 5 6
3 3

样例输出：
1 2 3 4 5 6

备注：
1 ≤ n ≤ 1000，1 ≤ pos ≤ n+1，元素值为整数。位置从 1 开始计数。

答案:
【C】
```c
#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    int a[1005];
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    int pos, x;
    scanf("%d %d", &pos, &x);
    if (pos < 1 || pos > n + 1) {
        printf("ERROR");
        return 0;
    }

    for (int i = n; i >= pos; i--) {
        a[i] = a[i - 1];
    }
    a[pos - 1] = x;
    n++;

    for (int i = 0; i < n; i++) {
        if (i > 0) printf(" ");
        printf("%d", a[i]);
    }
    return 0;
}
```

【C++】
```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; i++) cin >> a[i];

    int pos, x;
    cin >> pos >> x;
    if (pos < 1 || pos > n + 1) {
        cout << "ERROR";
        return 0;
    }

    a.insert(a.begin() + pos - 1, x);
    for (int i = 0; i < (int)a.size(); i++) {
        if (i > 0) cout << " ";
        cout << a[i];
    }
    return 0;
}
```

【Python】
```python
n = int(input())
a = list(map(int, input().split()))
pos, x = map(int, input().split())

if pos < 1 or pos > n + 1:
    print("ERROR")
else:
    a.insert(pos - 1, x)
    print(" ".join(map(str, a)))
```
解析: 顺序表采用连续存储结构，在第 pos 个位置插入元素时，需要将 pos 及其后面的元素整体向后移动一位，再把新元素放入 pos-1 下标处。最坏情况下需要移动 n 个元素，时间复杂度为 O(n)。
时间限制: 60
排序: 1

=== 知识点ID: 12
=== 知识点名称: 链表

## PROGRAMMING
题型: PROGRAMMING
难度: EASY
类别: BASIC
题目: 单链表反转
描述:
给定一个单链表中的 n 个整数，请反转链表并输出反转后的元素序列。

输入说明：
第一行输入整数 n，表示链表节点个数。
第二行输入 n 个整数，表示链表从头到尾的节点值。

输出说明：
输出反转后的链表元素，元素之间用一个空格分隔。
若 n 为 0，输出 EMPTY。

样例输入：
5
1 2 3 4 5

样例输出：
5 4 3 2 1

备注：
0 ≤ n ≤ 1000，节点值为整数。

答案:
【C】
```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

Node* createNode(int x) {
    Node* node = (Node*)malloc(sizeof(Node));
    node->data = x;
    node->next = NULL;
    return node;
}

int main() {
    int n;
    scanf("%d", &n);
    if (n == 0) {
        printf("EMPTY");
        return 0;
    }

    Node* head = NULL;
    Node* tail = NULL;
    for (int i = 0; i < n; i++) {
        int x;
        scanf("%d", &x);
        Node* node = createNode(x);
        if (head == NULL) {
            head = tail = node;
        } else {
            tail->next = node;
            tail = node;
        }
    }

    Node* prev = NULL;
    Node* cur = head;
    while (cur != NULL) {
        Node* next = cur->next;
        cur->next = prev;
        prev = cur;
        cur = next;
    }
    head = prev;

    Node* p = head;
    int first = 1;
    while (p != NULL) {
        if (!first) printf(" ");
        printf("%d", p->data);
        first = 0;
        Node* temp = p;
        p = p->next;
        free(temp);
    }
    return 0;
}
```

【C++】

```cpp
    }
    return 0;

```

【Python】
```python
n = int(input())
if n == 0:
    print("EMPTY")
else:
    values = list(map(int, input().split()))

    class Node:

            self.data = data
            self.next = None

        node = Node(x)
        if head is None:
            head = tail = node
        else:
            tail.next = node
            tail = node

    prev = None
    cur = head
    while cur is not None:
        nxt = cur.next
        cur.next = prev
        prev = cur
        cur = nxt
    head = prev

    result = []
    while head is not None:
        result.append(str(head.data))
        head = head.next
    print(" ".join(result))
```
解析: 单链表反转的核心是改变每个节点的 next 指针方向。遍历链表时保存当前节点的后继，再让当前节点指向前驱，最后将前驱和当前指针整体后移。每个节点只访问一次，时间复杂度 O(n)，额外空间复杂度 O(1)。
时间限制: 60
排序: 1

=== 知识点ID: 13
=== 知识点名称: 栈

## PROGRAMMING
题型: PROGRAMMING
难度: EASY
类别: BASIC
题目: 有效的括号
描述:
给定一个只包含 '('，')'，'['，']'，'{'，'}' 的字符串，判断括号是否匹配。

输入说明：
输入一行字符串 s。

输出说明：
如果所有括号都能正确匹配，输出 true；否则输出 false。

样例输入：
{[]}

样例输出：
true

备注：
1 ≤ 字符串长度 ≤ 1000。

答案:
【C】
```c
#include <stdio.h>
#include <string.h>

int main() {
    char s[1005];
    scanf("%s", s);
    char stack[1005];
    int top = -1;

    for (int i = 0; s[i] != '\0'; i++) {
        char ch = s[i];
        if (ch == '(' || ch == '[' || ch == '{') {
            stack[++top] = ch;
        } else {
            if (top == -1) {
                printf("false");
                return 0;
            }
            char left = stack[top--];
            if ((ch == ')' && left != '(') ||
                (ch == ']' && left != '[') ||
                (ch == '}' && left != '{')) {
                printf("false");
                return 0;
            }
        }
    }

    printf(top == -1 ? "true" : "false");
    return 0;
}
```

【C++】
```cpp
#include <iostream>
#include <stack>
#include <string>
#include <unordered_map>
using namespace std;

int main() {
    string s;
    cin >> s;
    stack<char> st;
    unordered_map<char, char> match = {
        {')', '('},
        {']', '['},
        {'}', '{'}
    };

    for (char ch : s) {
        if (ch == '(' || ch == '[' || ch == '{') {
            st.push(ch);
        } else {
            if (st.empty() || st.top() != match[ch]) {
                cout << "false";
                return 0;
            }
            st.pop();
        }
    }

    cout << (st.empty() ? "true" : "false");
    return 0;
}
```

【Python】

解析: 左括号入栈，遇到右括号时检查栈顶是否为对应的左括号。若栈为空或类型不匹配，则括号序列非法。遍历结束后栈为空，说明所有左括号都被正确匹配。

=== 知识点ID: 14
=== 知识点名称: 队列

## PROGRAMMING
题型: PROGRAMMING
难度: MEDIUM
类别: BASIC
题目: 循环队列操作模拟
描述:
给定循环队列的容量和一组操作，请按顺序模拟入队、出队、取队首和查询长度操作。

输入说明：
第一行输入两个整数 capacity 和 q，分别表示队列容量和操作次数。
接下来 q 行，每行表示一个操作：
ENQ x 表示将 x 入队；
DEQ 表示出队并输出出队元素；
FRONT 表示输出队首元素；
SIZE 表示输出当前队列长度。

输出说明：
对每个需要输出的操作输出一行。
入队时若队列已满，输出 FULL。
出队或取队首时若队列为空，输出 EMPTY。

样例输入：
3 7
ENQ 1
ENQ 2
FRONT
DEQ
ENQ 3
SIZE
DEQ

样例输出：
1
1
2
2

备注：
1 ≤ capacity ≤ 1000，1 ≤ q ≤ 1000，入队元素为整数。

答案:
【C】
```c
#include <stdio.h>
#include <string.h>

int main() {
    int capacity, q;
    scanf("%d %d", &capacity, &q);
    int queue[1005];
    int front = 0, rear = 0, size = 0;

    for (int i = 0; i < q; i++) {
        char op[10];
        scanf("%s", op);
        if (strcmp(op, "ENQ") == 0) {
            int x;
            scanf("%d", &x);
            if (size == capacity) {
                printf("FULL\n");
            } else {
                queue[rear] = x;
                rear = (rear + 1) % capacity;
                size++;
            }
        } else if (strcmp(op, "DEQ") == 0) {
            if (size == 0) {
                printf("EMPTY\n");
            } else {
                printf("%d\n", queue[front]);
                front = (front + 1) % capacity;
                size--;
            }
        } else if (strcmp(op, "FRONT") == 0) {
            if (size == 0) {
                printf("EMPTY\n");
            } else {
                printf("%d\n", queue[front]);
            }
        } else if (strcmp(op, "SIZE") == 0) {
            printf("%d\n", size);
        }
    }
    return 0;
}
```

【C++】
```cpp
#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main() {
    int capacity, q;

    vector<int> queue(capacity);

            if (size == 0) {
                cout << "EMPTY\n";
            } else {
                cout << queue[front] << "\n";

                size--;
            }
        } else if (op == "FRONT") {
            if (size == 0) {
                cout << "EMPTY\n";
            } else {
                cout << queue[front] << "\n";
            }
        } else if (op == "SIZE") {
            cout << size << "\n";
        }
    }
    return 0;
}
```

【Python】
```python
capacity, q = map(int, input().split())
queue = [0] * capacity
front = 0
rear = 0
size = 0
out = []

for _ in range(q):
    parts = input().split()
    op = parts[0]
    if op == "ENQ":
        x = int(parts[1])
        if size == capacity:
            out.append("FULL")
        else:
            queue[rear] = x
            rear = (rear + 1) % capacity
            size += 1
    elif op == "DEQ":
        if size == 0:
            out.append("EMPTY")
        else:
            out.append(str(queue[front]))
            front = (front + 1) % capacity
            size -= 1
    elif op == "FRONT":
        if size == 0:
            out.append("EMPTY")
        else:
            out.append(str(queue[front]))
    elif op == "SIZE":
        out.append(str(size))

print("\n".join(out))
```
解析: 循环队列用 front 指向队首位置，rear 指向下一个可插入位置。入队后 rear 按取模方式后移，出队后 front 按取模方式后移，从而让数组空间循环使用。本题额外维护 size 区分队空和队满，队空条件为 size == 0，队满条件为 size == capacity。
时间限制: 60
排序: 1
