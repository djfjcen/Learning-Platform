## PROGRAMMING
题型: PROGRAMMING
难度: EASY
类别: BASIC
题目: 集合去重
描述:
给定一个包含重复元素的整数序列，请去除重复元素后按升序输出。

输入说明：
第一行输入一个整数 n，表示序列中元素的个数。
第二行输入 n 个整数，以空格分隔。

输出说明：
输出去重后按升序排列的元素，以空格分隔。

样例输入：
8
3 1 2 3 5 2 1 4

样例输出：
1 2 3 4 5

备注：
1 ≤ n ≤ 1000，元素值范围 0~10000。

答案:
【C】
```c
#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    int exists[10001] = {0};
    for (int i = 0; i < n; i++) {
        int x;
        scanf("%d", &x);
        exists[x] = 1;
    }
    int first = 1;
    for (int i = 0; i <= 10000; i++) {
        if (exists[i]) {
            if (!first) printf(" ");
            printf("%d", i);
            first = 0;
        }
    }
    printf("\n");
    return 0;
}
```

【C++】

## PROGRAMMING
题型: PROGRAMMING
难度: EASY
类别: BASIC
题目: 集合元素查找
描述:
给定一个整数集合 A 和若干查询值，判断每个查询值是否属于集合 A。

输入说明：
第一行输入两个整数 n 和 m，分别表示集合 A 的元素个数和查询次数。
第二行输入 n 个整数，表示集合 A 的元素。
接下来 m 行，每行一个整数 q，表示查询值。

输出说明：
对每个查询，若 q ∈ A 输出 "YES"，否则输出 "NO"，每行一个结果。

样例输入：
5 3
1 3 5 7 9
3
4
9

样例输出：
YES
NO
YES

备注：
1 ≤ n, m ≤ 100000，元素值范围 -10^9 ~ 10^9。

答案:
【C】
```c
#include <stdio.h>
#include <stdlib.h>

int cmp(const void* a, const void* b) {
    long long diff = *(long long*)a - *(long long*)b;
    return diff > 0 ? 1 : (diff < 0 ? -1 : 0);
}

int binarySearch(long long* arr, int n, long long key) {
    int left = 0, right = n - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] == key) return 1;
        else if (arr[mid] < key) left = mid + 1;
        else right = mid - 1;
    }
    return 0;
}

int main() {
    int n, m;
    scanf("%d %d", &n, &m);
    long long* arr = (long long*)malloc(n * sizeof(long long));
    for (int i = 0; i < n; i++) scanf("%lld", &arr[i]);
    qsort(arr, n, sizeof(long long), cmp);
    for (int i = 0; i < m; i++) {
        long long q;
        scanf("%lld", &q);
        printf("%s\n", binarySearch(arr, n, q) ? "YES" : "NO");
    }
    free(arr);
    return 0;
}
```

=== 知识点名称: 集合运算

## PROGRAMMING
题型: PROGRAMMING
难度: MEDIUM
类别: BASIC
题目: 集合运算
描述:
给定两个整数集合 A 和 B，分别计算它们的并集、交集和差集 A-B。

输入说明：
第一行输入两个整数 n 和 m，分别表示集合 A 和 B 的元素个数。
第二行输入 n 个整数，表示集合 A 的元素。
第三行输入 m 个整数，表示集合 B 的元素。

输出说明：
第一行输出并集 A ∪ B 的元素个数和元素（升序，空格分隔）。
第二行输出交集 A ∩ B 的元素个数和元素（升序，空格分隔）。
第三行输出差集 A - B 的元素个数和元素（升序，空格分隔）。
每行先输出元素个数，再输出元素。

样例输入：
4 3
1 2 3 4
3 4 5

样例输出：
5 1 2 3 4 5
2 3 4
2 1 2

备注：
1 ≤ n, m ≤ 100000，元素值范围 -10^9 ~ 10^9。集合中无重复元素。

答案:
【C】
```c
#include <stdio.h>
#include <stdlib.h>

long long A[100005], B[100005];
int cmp(const void* a, const void* b) {
    long long diff = *(long long*)a - *(long long*)b;
    return diff > 0 ? 1 : (diff < 0 ? -1 : 0);
}

int main() {
    int n, m;
    scanf("%d %d", &n, &m);
    for (int i = 0; i < n; i++) scanf("%lld", &A[i]);
    for (int i = 0; i < m; i++) scanf("%lld", &B[i]);
    qsort(A, n, sizeof(long long), cmp);
    qsort(B, m, sizeof(long long), cmp);

    long long u[200010];
    int uc = 0, i = 0, j = 0;
    while (i < n && j < m) {
        if (A[i] < B[j]) u[uc++] = A[i++];
        else if (A[i] > B[j]) u[uc++] = B[j++];
        else { u[uc++] = A[i++]; j++; }
    }
    while (i < n) u[uc++] = A[i++];
    while (j < m) u[uc++] = B[j++];
    printf("%d", uc);
    for (int k = 0; k < uc; k++) printf(" %lld", u[k]);
    printf("\n");

    long long inter[100005];
    int ic = 0; i = 0; j = 0;
    while (i < n && j < m) {
        if (A[i] < B[j]) i++;
        else if (A[i] > B[j]) j++;
        else { inter[ic++] = A[i++]; j++; }
    }
    printf("%d", ic);
    for (int k = 0; k < ic; k++) printf(" %lld", inter[k]);
    printf("\n");

    long long diff[100005];
    int dc = 0; i = 0; j = 0;
    while (i < n && j < m) {
        if (A[i] < B[j]) diff[dc++] = A[i++];
        else if (A[i] > B[j]) j++;
        else { i++; j++; }
    }
    while (i < n) diff[dc++] = A[i++];
    printf("%d", dc);
    for (int k = 0; k < dc; k++) printf(" %lld", diff[k]);
    printf("\n");

    return 0;
}
```

=== 知识点ID: 10
=== 知识点名称: 集合存储实现

## PROGRAMMING
题型: PROGRAMMING
难度: MEDIUM
类别: INTERVIEW
题目: 哈希集合实现
描述:
请实现一个简单的哈希集合，支持以下三种操作：
1. add(x) — 向集合中插入元素 x，若已存在则不操作
2. contains(x) — 判断元素 x 是否在集合中
3. remove(x) — 从集合中删除元素 x，若不存在则不操作

输入说明：
第一行输入操作次数 n。
接下来 n 行，每行格式为 "add x"、"contains x" 或 "remove x"。

输出说明：
对每个 contains 操作，输出 "true" 或 "false"，每行一个。

样例输入：
7
add 1
add 2
contains 1
contains 3
add 2
remove 1
contains 1

样例输出：
true
false
false

备注：
1 ≤ n ≤ 100000，元素值范围 0 ~ 10^6。

答案:
【C】
```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define SIZE 10007

typedef struct Node {
    int val;
    struct Node* next;
} Node;

Node* table[SIZE];

int hash(int x) { return ((unsigned)x * 2654435761u) % SIZE; }

void add(int x) {
    int h = hash(x);
    Node* cur = table[h];
    while (cur) {
        if (cur->val == x) return;
        cur = cur->next;
    }
    Node* node = (Node*)malloc(sizeof(Node));
    node->val = x;
    node->next = table[h];
    table[h] = node;
}

int contains(int x) {
    int h = hash(x);
    Node* cur = table[h];
    while (cur) {
        if (cur->val == x) return 1;
        cur = cur->next;
    }
    return 0;
}

void removeVal(int x) {
    int h = hash(x);
    Node** pp = &table[h];
    while (*pp) {
        if ((*pp)->val == x) {
            Node* tmp = *pp;
            *pp = tmp->next;
            free(tmp);
            return;
        }
        pp = &(*pp)->next;
    }
}

int main() {
    memset(table, 0, sizeof(table));
    int n;
    scanf("%d", &n);
    char op[10];
    int x;
    while (n--) {
        scanf("%s %d", op, &x);
        if (strcmp(op, "add") == 0) add(x);
        else if (strcmp(op, "contains") == 0) printf("%s\n", contains(x) ? "true" : "false");
        else if (strcmp(op, "remove") == 0) removeVal(x);
    }
    return 0;
}
```

【C++】
```cpp
#include <bits/stdc++.h>
using namespace std;

class MyHashSet {
    vector<list<int>> table;
    int cap;
    size_t hash(int x) { return ((size_t)x * 2654435761u) % cap; }
public:
    MyHashSet(int c = 10007) : cap(c), table(c) {}
    void add(int x) {
        size_t h = hash(x);
        for (int v : table[h]) if (v == x) return;
        table[h].push_back(x);
    }
    bool contains(int x) {
        size_t h = hash(x);
        for (int v : table[h]) if (v == x) return true;
        return false;
    }
    void remove(int x) {
        size_t h = hash(x);
        auto& bucket = table[h];
        for (auto it = bucket.begin(); it != bucket.end(); ++it) {
            if (*it == x) { bucket.erase(it); return; }
        }
    }
};

int main() {
    int n;
    cin >> n;
    MyHashSet s;
    string op;
    int x;
    while (n--) {
        cin >> op >> x;
        if (op == "add") s.add(x);
        else if (op == "contains") cout << (s.contains(x) ? "true" : "false") << endl;
        else if (op == "remove") s.remove(x);
    }
    return 0;
}
```
