## PROGRAMMING
题型: PROGRAMMING
难度: EASY
类别: BASIC
题目: 邻接表的构建与输出
描述:
给定 n 个顶点（编号 0 至 n-1）和 m 条无向边，请构建邻接表并输出每个顶点的邻接点列表（按升序排列）。

输入说明：
第一行输入两个整数 n 和 m，分别表示顶点个数和边的条数。
接下来 m 行，每行两个整数 u v，表示顶点 u 和 v 之间的一条无向边。

输出说明：
输出 n 行，第 i 行先输出顶点 i 及其邻接点的个数，再按升序输出所有邻接点。
格式为 "i: cnt adj1 adj2 ..."，具体见样例。

样例输入：
4 4
0 1
0 2
1 3
2 3

样例输出：
0: 2 1 2
1: 2 0 3
2: 2 0 3
3: 2 1 2

备注：
1 ≤ n ≤ 1000，0 ≤ m ≤ 10000。

答案:
【C】
```c
#include <stdio.h>
#include <stdlib.h>

int adj[1000][1000];
int deg[1000];

int cmp(const void* a, const void* b) {
    return *(int*)a - *(int*)b;
}

int main() {
    int n, m;
    scanf("%d %d", &n, &m);
    for (int i = 0; i < m; i++) {
        int u, v;
        scanf("%d %d", &u, &v);
        adj[u][deg[u]++] = v;
        adj[v][deg[v]++] = u;
    }
    for (int i = 0; i < n; i++) {
        qsort(adj[i], deg[i], sizeof(int), cmp);
        printf("%d: %d", i, deg[i]);
        for (int j = 0; j < deg[i]; j++)
            printf(" %d", adj[i][j]);
        printf("\n");
    }
    return 0;
}
```

=== 知识点名称: 图的遍历

## PROGRAMMING
题型: PROGRAMMING
难度: EASY
类别: BASIC
题目: 连通分量个数
描述:
给定一个无向图，请统计图中连通分量的个数。连通分量是指图中极大连通子图。

输入说明：
第一行输入两个整数 n 和 m，分别表示顶点个数和边的条数。
接下来 m 行，每行两个整数 u v，表示顶点 u 和 v 之间的一条无向边。

输出说明：
输出一个整数，表示连通分量的个数。

样例输入：
6 3
0 1
1 2
3 4

样例输出：
3

备注：
1 ≤ n ≤ 1000，0 ≤ m ≤ 10000。顶点编号为 0 至 n-1。

答案:
【C】
```c
#include <stdio.h>
#include <string.h>

int adj[1000][1000];
int deg[1000];
int visited[1000];
int n;

void dfs(int v) {
    visited[v] = 1;
    for (int i = 0; i < deg[v]; i++) {
        int u = adj[v][i];
        if (!visited[u])
            dfs(u);
    }
}

int main() {
    int m;
    scanf("%d %d", &n, &m);
    memset(deg, 0, sizeof(deg));
    memset(visited, 0, sizeof(visited));
    for (int i = 0; i < m; i++) {
        int u, v;
        scanf("%d %d", &u, &v);
        adj[u][deg[u]++] = v;
        adj[v][deg[v]++] = u;
    }
    int cnt = 0;
    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            dfs(i);
            cnt++;
        }
    }
    printf("%d\n", cnt);
    return 0;
}
```

```cpp
                }
            }
        }
    }

}
```

## PROGRAMMING
题型: PROGRAMMING
难度: MEDIUM
类别: BASIC
题目: 单源最短路径
描述:
给定一个 n 个顶点的带权无向图，求从源点 s 出发到所有顶点的最短距离。

输入说明：
第一行输入三个整数 n、m 和 s，分别表示顶点数、边数和源点编号。
接下来 m 行，每行三个整数 u v w，表示顶点 u 和 v 之间有一条权值为 w 的边。

输出说明：
输出一行 n 个整数，分别表示从源点 s 到顶点 0 至 n-1 的最短距离。若不可达则输出 -1。

样例输入：
5 6 0
0 1 4
0 2 2
1 2 1
1 3 5
2 3 8
2 4 10

样例输出：
0 3 2 8 12

备注：
1 ≤ n ≤ 1000，0 ≤ m ≤ 10000，0 ≤ w ≤ 1000。保证边权非负。

答案:
【C】
```c
#include <stdio.h>
#include <string.h>

#define INF 0x3f3f3f3f

int adj[1000][1000];
int weight[1000][1000];
int deg[1000];
int dist[1000];
int visited[1000];

int main() {
    int n, m, s;
    scanf("%d %d %d", &n, &m, &s);
    memset(deg, 0, sizeof(deg));
    for (int i = 0; i < m; i++) {
        int u, v, w;
        scanf("%d %d %d", &u, &v, &w);
        adj[u][deg[u]] = v;
        weight[u][deg[u]] = w;
        deg[u]++;
        adj[v][deg[v]] = u;
        weight[v][deg[v]] = w;
        deg[v]++;
    }
    memset(visited, 0, sizeof(visited));
    memset(dist, 0x3f, sizeof(dist));
    dist[s] = 0;
    for (int i = 0; i < n; i++) {
        int u = -1, minD = INF;
        for (int j = 0; j < n; j++) {
            if (!visited[j] && dist[j] < minD) {
                minD = dist[j];
                u = j;
            }
        }
        if (u == -1) break;
        visited[u] = 1;
        for (int j = 0; j < deg[u]; j++) {
            int v = adj[u][j];
            int w = weight[u][j];
            if (!visited[v] && dist[u] + w < dist[v])
                dist[v] = dist[u] + w;
        }
    }
    for (int i = 0; i < n; i++) {
        if (i) printf(" ");
        if (dist[i] >= INF / 2) printf("-1");
        else printf("%d", dist[i]);
    }
    printf("\n");
    return 0;
}
```

```cpp
#include <bits/stdc++.h>
using namespace std;

const int INF = 0x3f3f3f3f;

int main() {
    int n, m, s;
    cin >> n >> m >> s;
    vector<vector<pair<int, int>>> adj(n);
    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }
    vector<int> dist(n, INF);
    dist[s] = 0;
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<>> pq;
    pq.push({0, s});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;
        for (auto [v, w] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    for (int i = 0; i < n; i++) {
        if (i) cout << " ";
        if (dist[i] >= INF / 2) cout << -1;
        else cout << dist[i];
    }
    cout << endl;
    return 0;
}
```

```python
for _ in range(m):

while pq:

ans = []
for i in range(n):

print(' '.join(ans))
```

=== 知识点ID: 22
=== 知识点名称: 最小生成树

## PROGRAMMING
题型: PROGRAMMING
难度: MEDIUM
类别: BASIC
题目: 最小生成树权值和
描述:
给定一个 n 个顶点的带权无向连通图，求其最小生成树（MST）的权值和。

输入说明：
第一行输入两个整数 n 和 m，分别表示顶点数和边数。
接下来 m 行，每行三个整数 u v w，表示顶点 u 和 v 之间有一条权值为 w 的边。

输出说明：
输出一个整数，表示最小生成树的权值和。若图不连通，输出 -1。

样例输入：
5 7
0 1 2
0 3 6
0 4 4
1 2 3
1 3 8
2 3 7
3 4 5

样例输出：
14

备注：
1 ≤ n ≤ 1000，0 ≤ m ≤ 100000，边权为 int 范围内正整数。

答案:
【C】
```c
#include <stdio.h>
#include <stdlib.h>

typedef struct {
    int u, v, w;
} Edge;

Edge edges[100005];
int parent[1005];

int cmp(const void* a, const void* b) {
    return ((Edge*)a)->w - ((Edge*)b)->w;
}

int find(int x) {
    if (parent[x] != x)
        parent[x] = find(parent[x]);
    return parent[x];
}

int unite(int x, int y) {
    int rx = find(x), ry = find(y);
    if (rx == ry) return 0;
    parent[rx] = ry;
    return 1;
}

int main() {
    int n, m;
    scanf("%d %d", &n, &m);
    for (int i = 0; i < m; i++)
        scanf("%d %d %d", &edges[i].u, &edges[i].v, &edges[i].w);
    qsort(edges, m, sizeof(Edge), cmp);
    for (int i = 0; i < n; i++) parent[i] = i;
    int sum = 0, cnt = 0;
    for (int i = 0; i < m && cnt < n - 1; i++) {
        if (unite(edges[i].u, edges[i].v)) {
            sum += edges[i].w;
            cnt++;
        }
    }
    printf("%d\n", cnt == n - 1 ? sum : -1);
    return 0;
}
```

【C++】
```cpp
#include <bits/stdc++.h>
using namespace std;

struct Edge {
    int u, v, w;
    bool operator<(const Edge& other) const {
        return w < other.w;
    }
};

class UnionFind {
    vector<int> parent;
public:
    UnionFind(int n) : parent(n) {
        for (int i = 0; i < n; i++) parent[i] = i;
    }
    int find(int x) {
        if (parent[x] != x)
            parent[x] = find(parent[x]);
        return parent[x];
    }
    bool unite(int x, int y) {
        int rx = find(x), ry = find(y);
        if (rx == ry) return false;
        parent[rx] = ry;
        return true;
    }
};

int main() {
    int n, m;
    cin >> n >> m;
    vector<Edge> edges(m);
    for (int i = 0; i < m; i++)
        cin >> edges[i].u >> edges[i].v >> edges[i].w;
    sort(edges.begin(), edges.end());
    UnionFind uf(n);
    int sum = 0, cnt = 0;
    for (const auto& e : edges) {
        if (uf.unite(e.u, e.v)) {
            sum += e.w;
            if (++cnt == n - 1) break;
        }
    }
    cout << (cnt == n - 1 ? sum : -1) << endl;
    return 0;
}
```

【Python】
```python
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
    
    def find(self, x):
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]
    
    def unite(self, x, y):
        rx, ry = self.find(x), self.find(y)
        if rx == ry:
            return False
        self.parent[rx] = ry
        return True

n, m = map(int, input().split())
edges = []
for _ in range(m):
    u, v, w = map(int, input().split())
    edges.append((w, u, v))

edges.sort()
uf = UnionFind(n)
total = cnt = 0
for w, u, v in edges:
    if uf.unite(u, v):
        total += w
        cnt += 1
        if cnt == n - 1:
            break

print(total if cnt == n - 1 else -1)
```
