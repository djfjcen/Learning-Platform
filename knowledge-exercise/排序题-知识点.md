=== 知识点ID: 27
=== 知识点名称: 直接插入排序

## OPERATION
标题: 直接插入排序
内容:
将序列划分为已排序区和未排序区。
每次从无序区取一个元素，在有序区从后向前扫描并插入合适位置。
最好情况：序列已有序，仅需比较 n-1 次。
最坏情况：序列逆序，每个元素都需移动。

## COMPLEXITY
标题: 复杂度分析
内容:
- 最好时间复杂度：O(n)
- 最坏时间复杂度：O(n²)
- 平均时间复杂度：O(n²)
- 空间复杂度：O(1)
- 稳定性：稳定排序

## CODE
标题: 直接插入排序实现
描述: 对数组进行升序排列
语言: C++
代码:
#include <bits/stdc++.h>
using namespace std;

void insertionSort(vector<int>& a) {
    for (int i = 1; i < a.size(); i++) {
        int key = a[i], j = i - 1;
        while (j >= 0 && a[j] > key) {
            a[j + 1] = a[j];
            j--;
        }
        a[j + 1] = key;
    }
}

int main() {
    vector<int> a = {5, 2, 9, 1, 3};
    insertionSort(a);
    for (int x : a) cout << x << " ";
    return 0;
}

=== 知识点ID: 27
=== 知识点名称: 折半插入排序

## OPERATION
标题: 折半插入排序
内容:
在直接插入排序的基础上，使用二分查找确定插入位置。
减少了关键字比较次数，但元素移动次数不变。
比较次数约为 O(n log n)，移动次数仍为 O(n²)。

## COMPLEXITY
标题: 复杂度分析
内容:
- 时间复杂度：O(n²)
- 比较次数：O(n log n)
- 移动次数：O(n²)
- 空间复杂度：O(1)
- 稳定性：稳定排序

## CODE
标题: 折半插入排序实现
描述: 使用二分查找优化插入位置
语言: C++
代码:
#include <bits/stdc++.h>
using namespace std;

void binaryInsertionSort(vector<int>& a) {
    int n = a.size();
    for (int i = 1; i < n; i++) {
        int key = a[i];
        int l = 0, r = i - 1;
        while (l <= r) {
            int mid = l + (r - l) / 2;
            if (a[mid] > key) r = mid - 1;
            else l = mid + 1;
        }
        for (int j = i - 1; j >= l; j--)
            a[j + 1] = a[j];
        a[l] = key;
    }
}

int main() {
    vector<int> a = {5, 2, 9, 1, 3};
    binaryInsertionSort(a);
    for (int x : a) cout << x << " ";
    return 0;
}

标题: 链表插入排序实现
描述: 对单链表进行升序排序
语言: C++
代码:
#include <bits/stdc++.h>
using namespace std;

struct ListNode {
    int val;
    ListNode* next;
    ListNode(int x) : val(x), next(nullptr) {}
};

ListNode* insertionSortList(ListNode* head) {
    ListNode dummy(0);
    while (head) {
        ListNode* cur = head;
        head = head->next;
        ListNode* prev = &dummy;
        while (prev->next && prev->next->val < cur->val)
            prev = prev->next;
        cur->next = prev->next;
        prev->next = cur;
    }
    return dummy.next;
}

- 稳定性：不稳定排序

## CODE
标题: 希尔排序实现
描述: 使用 Shell 增量
语言: C++
代码:
#include <bits/stdc++.h>
using namespace std;

void shellSort(vector<int>& a) {
    int n = a.size();
    for (int gap = n / 2; gap > 0; gap /= 2) {
        for (int i = gap; i < n; i++) {
            int t = a[i], j = i;
            while (j >= gap && a[j - gap] > t) {
                a[j] = a[j - gap];
                j -= gap;
            }
            a[j] = t;
        }
    }
}

int main() {
    vector<int> a = {9, 1, 5, 8, 3, 7, 4, 6, 2};
    shellSort(a);
    for (int x : a) cout << x << " ";
    return 0;
}

#include <bits/stdc++.h>
using namespace std;

void bubbleSort(vector<int>& a) {
    int n = a.size();
    bool swapped;
    for (int i = 0; i < n - 1; i++) {
        swapped = false;
        for (int j = 0; j < n - i - 1; j++) {
            if (a[j] > a[j + 1]) {
                swap(a[j], a[j + 1]);
                swapped = true;
            }
        }
        if (!swapped) break;
    }
}

## OPERATION
标题: 快速排序
内容:
选基准元素 pivot。
通过划分将小于 pivot 的放左边，大于的放右边。
递归处理左右子序列。
划分方式影响性能。

## COMPLEXITY
标题: 复杂度分析
内容:
- 最好时间复杂度：O(n log n)
- 最坏时间复杂度：O(n²)
- 平均时间复杂度：O(n log n)
- 空间复杂度：递归栈 O(log n) ～ O(n)
- 稳定性：不稳定排序

## CODE
标题: 快速排序实现
描述: 递归版快速排序
语言: C++
代码:
#include <bits/stdc++.h>
using namespace std;

int partition(vector<int>& a, int l, int r) {
    int pivot = a[l], i = l;
    for (int j = l + 1; j <= r; j++) {
        if (a[j] < pivot)
            swap(a[++i], a[j]);
    }
    swap(a[l], a[i]);
    return i;
}

void quickSort(vector<int>& a, int l, int r) {
    if (l >= r) return;
    int p = partition(a, l, r);
    quickSort(a, l, p - 1);
    quickSort(a, p + 1, r);
}

int main() {
    vector<int> a = {6, 3, 8, 2, 7};
    quickSort(a, 0, a.size() - 1);
    for (int x : a) cout << x << " ";
    return 0;
}

=== 知识点名称: 三数取中快速排序

## OPERATION
标题: 三数取中快速排序
内容:
选择左、中、右三个元素的中值作为枢轴。
避免对已排序序列产生最坏情况。
划分采用双向扫描，减少交换次数。

## COMPLEXITY
标题: 复杂度分析
内容:
- 最坏时间复杂度：O(n²)（概率极低）
- 平均时间复杂度：O(n log n)
- 空间复杂度：递归栈 O(log n)
- 稳定性：不稳定排序

## CODE
标题: 三数取中快速排序实现
描述: 降低退化概率
语言: C++
代码:
#include <bits/stdc++.h>
using namespace std;

int medianPivot(vector<int>& a, int l, int r) {
    int m = l + (r - l) / 2;
    if (a[l] > a[m]) swap(a[l], a[m]);
    if (a[l] > a[r]) swap(a[l], a[r]);
    if (a[m] > a[r]) swap(a[m], a[r]);
    return a[m];
}

int partition(vector<int>& a, int l, int r) {
    int pivot = medianPivot(a, l, r);
    int i = l, j = r;
    while (true) {
        while (a[i] < pivot) i++;
        while (a[j] > pivot) j--;
        if (i >= j) return j;
        swap(a[i++], a[j--]);
    }
}

#include <bits/stdc++.h>
using namespace std;

void countingSort(vector<int>& a) {
    int mx = *max_element(a.begin(), a.end());
    vector<int> cnt(mx + 1, 0);
    for (int x : a) cnt[x]++;
    for (int i = 1; i <= mx; i++) cnt[i] += cnt[i - 1];
    vector<int> res(a.size());
    for (int i = a.size() - 1; i >= 0; i--)
        res[--cnt[a[i]]] = a[i];
    a = res;
}

#include <bits/stdc++.h>
using namespace std;

void selectionSort(vector<int>& a) {
    int n = a.size();
    for (int i = 0; i < n - 1; i++) {
        int minIdx = i;
        for (int j = i + 1; j < n; j++) {
            if (a[j] < a[minIdx])
                minIdx = j;
        }
        swap(a[i], a[minIdx]);
    }
}

#include <bits/stdc++.h>
using namespace std;

void heapify(vector<int>& a, int n, int i) {
    int largest = i, l = 2 * i + 1, r = 2 * i + 2;
    if (l < n && a[l] > a[largest]) largest = l;
    if (r < n && a[r] > a[largest]) largest = r;
    if (largest != i) {
        swap(a[i], a[largest]);
        heapify(a, n, largest);
    }
}

void heapSort(vector<int>& a) {
    int n = a.size();
    for (int i = n / 2 - 1; i >= 0; i--)
        heapify(a, n, i);
    for (int i = n - 1; i > 0; i--) {
        swap(a[0], a[i]);
        heapify(a, i, 0);
    }
}

int main() {
    vector<int> a = {4, 10, 3, 5, 1};
    heapSort(a);
    for (int x : a) cout << x << " ";
    return 0;
}

#include <bits/stdc++.h>
using namespace std;

void bucketSort(vector<float>& a) {
    int n = a.size();
    vector<vector<float>> b(n);
    for (int i = 0; i < n; i++)
        b[(int)(n * a[i])].push_back(a[i]);
    for (int i = 0; i < n; i++)
        sort(b[i].begin(), b[i].end());
    int idx = 0;
    for (int i = 0; i < n; i++)
        for (float x : b[i])
            a[idx++] = x;
}

#include <bits/stdc++.h>
using namespace std;

void merge(vector<int>& a, int l, int m, int r) {
    vector<int> t;
    int i = l, j = m + 1;
    while (i <= m && j <= r) {
        if (a[i] <= a[j]) t.push_back(a[i++]);
        else t.push_back(a[j++]);
    }
    while (i <= m) t.push_back(a[i++]);
    while (j <= r) t.push_back(a[j++]);
    for (int k = 0; k < t.size(); k++)
        a[l + k] = t[k];
}

void mergeSort(vector<int>& a, int l, int r) {
    if (l >= r) return;
    int m = l + (r - l) / 2;
    mergeSort(a, l, m);
    mergeSort(a, m + 1, r);
    merge(a, l, m, r);
}

int main() {
    vector<int> a = {5, 2, 4, 6, 1, 3};
    mergeSort(a, 0, a.size() - 1);
    for (int x : a) cout << x << " ";
    return 0;
}

#include <bits/stdc++.h>
using namespace std;

void radixSort(vector<int>& a) {
    int mx = *max_element(a.begin(), a.end());
    for (int exp = 1; mx / exp > 0; exp *= 10) {
        vector<int> output(a.size()), cnt(10, 0);
        for (int x : a) cnt[(x / exp) % 10]++;
        for (int i = 1; i < 10; i++) cnt[i] += cnt[i - 1];
        for (int i = a.size() - 1; i >= 0; i--)
            output[--cnt[(a[i] / exp) % 10]] = a[i];
        a = output;
    }
}
