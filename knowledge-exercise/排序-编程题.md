```c
#include <stdio.h>

void quickSort(int a[], int left, int right) {
    if (left >= right) return;
    int pivot = a[(left + right) / 2];
    int i = left, j = right;
    while (i <= j) {
        while (a[i] < pivot) i++;
        while (a[j] > pivot) j--;
        if (i <= j) {
            int t = a[i]; a[i] = a[j]; a[j] = t;
            i++; j--;
        }
    }
    quickSort(a, left, j);
    quickSort(a, i, right);
}

int main() {
    int n, a[100005];

    for (int i = 0; i < n; i++) scanf("%d", &a[i]);

    for (int i = 0; i < n; i++) {
        if (i) printf(" ");
        printf("%d", a[i]);
    }
    return 0;
}
```

```c
#include <stdio.h>

int selectKth(int a[], int left, int right, int k) {
    if (left == right) return a[left];
    int pivot = a[(left + right) / 2];
    int i = left, j = right;
    while (i <= j) {
        while (a[i] < pivot) i++;
        while (a[j] > pivot) j--;
        if (i <= j) {
            int t = a[i]; a[i] = a[j]; a[j] = t;
            i++; j--;
        }
    }

    if (k >= i) return selectKth(a, i, right, k);
    return a[k];
}
```

## PROGRAMMING
题型: PROGRAMMING
难度: HARD
类别: INTERVIEW
题目: 逆序对计数
描述:
给定 n 个整数，请统计其中逆序对的数量。逆序对定义：若 i < j 且 a[i] > a[j]，则 (i, j) 为一个逆序对。

输入说明：
第一行输入一个整数 n，表示元素个数。
第二行输入 n 个整数。

输出说明：
输出逆序对的数量。

样例输入：
5
3 1 4 1 5

样例输出：
3

备注：
1 <= n <= 100000，所有元素均为 int 范围内的整数。

答案:

```c
#include <stdio.h>

long long merge(int a[], int tmp[], int left, int mid, int right) {
    int i = left, j = mid + 1, k = left;
    long long count = 0;
    while (i <= mid && j <= right) {
        if (a[i] <= a[j]) {
            tmp[k++] = a[i++];
        } else {
            count += mid - i + 1;
            tmp[k++] = a[j++];
        }
    }
    while (i <= mid) tmp[k++] = a[i++];
    while (j <= right) tmp[k++] = a[j++];
    for (int p = left; p <= right; p++) a[p] = tmp[p];
    return count;
}

long long mergeSort(int a[], int tmp[], int left, int right) {
    if (left >= right) return 0;
    int mid = left + (right - left) / 2;
    long long count = 0;
    count += mergeSort(a, tmp, left, mid);
    count += mergeSort(a, tmp, mid + 1, right);
    count += merge(a, tmp, left, mid, right);
    return count;
}

    int n;
    scanf("%d", &n);
    int a[100005], tmp[100005];
    for (int i = 0; i < n; i++) scanf("%d", &a[i]);
```

```python
def merge_sort(a):
    if len(a) <= 1:
        return a, 0
    mid = len(a) // 2
    left, cnt_l = merge_sort(a[:mid])
    right, cnt_r = merge_sort(a[mid:])
    merged = []
    i = j = 0
    cnt = cnt_l + cnt_r
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            merged.append(left[i])
            i += 1
        else:
            cnt += len(left) - i
            merged.append(right[j])
            j += 1
    merged.extend(left[i:])
    merged.extend(right[j:])
    return merged, cnt

n = int(input())

print(ans)
```
